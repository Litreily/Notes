Shell

sudo

sudo su  # 进入root模式
sudo {cmd} # 以管理员身份执行指令

cd

cd  # 回到默认目录 ~ ,即/home/user-name
cd ..  # 回退到前级目录
cd {folder-path}  # 到达指定目录

ls

ls  # 显示当前路径下的文件及文件夹
ls -a  # 显示所有文件（包含隐藏文件及文件夹）

pwd

pwd  # 显示当前路径

mkdir

mkdir {folder-name}  # 新建文件夹

rm

rm {file-name}  # 删除文件
rm -rf {folder-name}  # 删除文件夹
